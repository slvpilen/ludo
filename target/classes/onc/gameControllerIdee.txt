import javafx.fxml.FXML;
import javafx.scene.input.MouseEvent;
import javafx.scene.shape.Circle;

public class MyController {
    @FXML
    private GridPane gridPane;
    @FXML
    private Circle circle1;
    @FXML
    private Circle circle2;
    // Add 14 more @FXML Circle fields

    // Add a method to handle mouse clicks on any circle
    @FXML
    private void handleCircleClick(MouseEvent event) {
        Circle clickedCircle = (Circle) event.getSource();
        int currentRow = GridPane.getRowIndex(clickedCircle);
        GridPane.setRowIndex(clickedCircle, currentRow + 2);
    }

    // Initialize the controller by adding the event handler to each circle
    @FXML
    private void initialize() {
        for (int i = 0; i < 16; i++) {
            Circle circle = getCircleById("circle" + (i+1));
            circle.setOnMouseClicked(this::handleCircleClick);
        }
    }

    // Helper method to get a circle by fx:id
    private Circle getCircleById(String id) {
        for (Node node : gridPane.getChildren()) {
            if (node instanceof Circle && node.getId().equals(id)) {
                return (Circle) node;
            }
        }
        return null;
    }
}
